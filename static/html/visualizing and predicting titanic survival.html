def compare_missing_columns(df, col_name, missing_values):
    a = df[col_name][missing_values]
    b = df[col_name][~missing_values]
    ttest = ttest_ind(a, b, nan_policy='omit') 
    pvalue = ttest.pvalue
    
    if pvalue < 0.05:    
        print(col_name, pvalue)
